## 描述

QTreeWidget中实现右键不同的节点显示不同的右键菜单
## 内容

```cpp
#include <QTreeWidget>
#include <QMenu>
#include <QAction>
#include <QMessageBox>

class MyTreeWidget : public QTreeWidget {
    Q_OBJECT

public:
    MyTreeWidget(QWidget *parent = nullptr) : QTreeWidget(parent) {
        // 启用自定义右键菜单
        setContextMenuPolicy(Qt::CustomContextMenu);
        
        // 连接信号槽
        connect(this, &QTreeWidget::customContextMenuRequested, this, &MyTreeWidget::showContextMenu);
    }

private slots:
    void showContextMenu(const QPoint &pos) {
        // 获取右键点击的节点
        QTreeWidgetItem *item = itemAt(pos);
        if (!item) return;

        // 创建右键菜单
        QMenu contextMenu(this);

        // 根据节点的类型或者内容来显示不同的菜单
        if (item->text(0) == "节点1") {
            QAction *action1 = new QAction("操作1", this);
            connect(action1, &QAction::triggered, this, []() {
                QMessageBox::information(nullptr, "Action", "操作1被点击");
            });
            contextMenu.addAction(action1);
        } else if (item->text(0) == "节点2") {
            QAction *action2 = new QAction("操作2", this);
            connect(action2, &QAction::triggered, this, []() {
                QMessageBox::information(nullptr, "Action", "操作2被点击");
            });
            contextMenu.addAction(action2);
        } else {
            QAction *defaultAction = new QAction("默认操作", this);
            connect(defaultAction, &QAction::triggered, this, []() {
                QMessageBox::information(nullptr, "Action", "默认操作被点击");
            });
            contextMenu.addAction(defaultAction);
        }

        // 在鼠标点击处显示菜单
        contextMenu.exec(mapToGlobal(pos));
    }
};

```
