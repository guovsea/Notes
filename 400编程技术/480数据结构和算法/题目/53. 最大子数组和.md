## 解答

```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int maxSum = -1e9;
        int currMaxSum = -1e9;
        for (int i = 0; i < nums.size(); i++) {
            // 要么将nums[i]加入到所考虑的连续数组中，要么从nums[i]重新开始
            currMaxSum = max(currMaxSum + nums[i], nums[i]);
            // 记录答案
            maxSum = max(maxSum, currMaxSum);
        }
        return maxSum;
    }
};
```
