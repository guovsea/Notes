使用多进程并发服务器时要考虑以下几点：

1. 父进程最大文件描述个数
2. 系统内创建进程个数(与内存大小相关)
3. 进程创建过多是否降低整体服务性能 (进程调度)
4. 父进程中需要关闭 connfd 返回的新文件描文件描述符；子进程中需要关闭 listenfd


## server

```c
#include <arpa/inet.h>
#include <ctype.h>
#include <errno.h>
#include <netinet/in.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <sys/wait.h>
#include <unistd.h>

#define BUFFER_SIZE 1024
#define PORT 8090
#define SERV_ADDR "192.168.10.10"
#define MAXLISTEN 1024

void sys_err(const char *str) {
    perror(str);
    exit(1);
}

void do_sigchild(int signum) {
    printf("SIGCHLD action");
    while (waitpid(0, NULL, WNOHANG) > 0)
        ;
}

int main(int argc, char *argv[]) {

    char buf[BUFFER_SIZE];
    int listenfd, connfd;

    struct sigaction newact;
    newact.sa_handler = do_sigchild;
    sigemptyset(&newact.sa_mask);
    newact.sa_flags = 0;
    sigaction(SIGCHLD, &newact, NULL);

    listenfd = socket(AF_INET, SOCK_STREAM, 0);
    if (listenfd < 0)
        sys_err("socket error");

    struct sockaddr_in serv_addr;

    serv_addr.sin_family = AF_INET;
    inet_pton(AF_INET, SERV_ADDR, &serv_addr.sin_addr);
    serv_addr.sin_port = htons(PORT);
        if (bind(listenfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0)
        sys_err("bind error");

    listen(listenfd, MAXLISTEN);
    struct sockaddr_in client_addr;
    socklen_t client_len;
    int pid;
    char client_str[INET_ADDRSTRLEN];
    int readn;

    printf("Server address = %s, port = %d\n", SERV_ADDR, PORT);
    printf("Accepting connection...\n");
    while (1) {
        client_len = sizeof(client_addr);
        connfd = accept(listenfd, (struct sockaddr *)&client_addr, &client_len);
        if (connfd < 0) {
            if (errno == EINTR) {
                continue;
            }
            sys_err("accept error");
        }
        pid = fork();
        if (pid > 0) {
            close(connfd);
        } else if (pid == 0) {
            close(listenfd);
            inet_ntop(AF_INET, &client_addr.sin_addr, client_str,
                      sizeof(client_str));
            printf("%s:%d connected\n", client_str,
                   ntohs(client_addr.sin_port));
            while (1) {
                if ((readn = read(connfd, buf, sizeof(buf))) < 0) {
                    sys_err("read from connfd error");
                } else if (readn == 0) {
                    printf("%s:%d dis connected\n", client_str,
                           ntohs(client_addr.sin_port));
                    break;
                } else {
                    for (int i = 0; i < readn; i++) {
                        buf[i] = toupper(buf[i]);
                    }
                    write(connfd, buf, readn);
                }
            }
            close(connfd);
            return 0;
        } else {
            sys_err("fork error");
        }
    }
    close(listenfd);
    return EXIT_SUCCESS;
}
```

**几点注意**

1. `accept`是慢速系统调用，会被信号捕捉中断[[慢速系统调用中断和 EINTER]]，因此需要重启
2. fork 后子进程要关闭 listenfd，父进程要关闭 connfd。
3. 信号处理函数中要使用`while(waitpid(0, NULL, WNOHANG) > 0) ;`循环回收所有子进程
