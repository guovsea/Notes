```cpp
    ListNode* mergeTwoLists(ListNode* L1, ListNode* L2) {
        if (!L1)  return L2;
        if (!L2)  return L1;
        ListNode* head = L1;
        ListNode *next1;
        ListNode *next2;
        while (L1 && L2) {
            next1 = L1->next;
            next2 = L2->next;
            L2->next = next1;
            L1->next = L2;
            L1 = next1;
            L2 = next2;
        }
        return head;
    }
```
