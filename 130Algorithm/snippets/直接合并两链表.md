```cpp
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode *dummy = new ListNode();
        ListNode* p = dummy;
        while (list1 != nullptr && list2 != nullptr) {
            if (list1->val < list2->val) {
                p->next = list1;
                list1 = list1->next;
            } else {
                p->next = list2;
                list2 = list2->next;
            }
            p = p->next;
        }
        // 未合并被合并完的链表直接连接到已经合并的链表的后面
        p->next = list1 !=nullptr ? list1 : list2;
        return dummy->next;
    }
```
