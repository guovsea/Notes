#include <iostream>
#include <string>
#include <vector>
using namespace std;

class Solution {
public:
    vector<string> restoreIpAddresses(string s) {
        vector<string> path;
        vector<string> res;
        dfs(s, s.length(), 0, 0, path, res);
        return res;
    }
    // 字符串，字符串长度，已有几段ip地址，下一段ip地址开始位置，回溯常规变量path，答案
    void dfs(const string s, int len, int split, int begin, vector<string> path, vector<string>& res) {
        if (begin == len) {
            if (split == 4) {
                string ip = "";
                for (string s : path) {
                    ip += s;
                    ip += ".";
                }
                ip.pop_back();
                res.push_back(ip);
            }
            return;
        }
        // 如果剩余字符串长度(left - begin)不足够进行split，或者超过能split的最大长度
        if (len - begin < (4 - split) || len - begin > 3 * (4 - split)) {
            return;  // 则直接终止，不再继续向下搜索
        }
        // 下一段ip的长度是多少(1~3)
        for (int i = 1; i <= 3; i++) {
            if (begin + i > len) {
                return; // 字符串越界，不能搜索该i值
            }
            int ipSetment = ToIpSegment(s, begin, i);
            if (ipSetment != -1) {
                string test = s.substr(begin, i);
                path.push_back(s.substr(begin, i));
                dfs(s, len, split + 1, begin + i, path, res);
                path.pop_back(); // 回溯，恢复原来的状态
            }
        }
    }
    int ToIpSegment(string s, int left, int len) {
        if (len > 1 && s[left] == '0')
            return -1; // 大于 1 位不能以'0'开头
        int res = 0;
        for (int i = 0; i < len; i++) {
            // 这种字符串转数字方式很巧妙
            res = res * 10 + s[left + i] - '0';
        }
        return res > 255 ? -1 : res;
    }
};

int main(int argc, char const *argv[]) {
    Solution solution;
    string s ="101023";
    cout << solution.ToIpSegment(s, 2, 3) << endl;
    cout << s.substr(2, 3) << endl;
    vector<string> res;
    res = solution.restoreIpAddresses(s);
    for (string s : res) {
        cout << s  << endl;
    }
    return 0;
}

## 题目

**有效 IP 地址** 正好由四个整数（每个整数位于 `0` 到 `255` 之间组成，且不能含有前导 `0`），整数之间用 `'.'` 分隔。

- 例如：`"0.1.2.201"` 和 `"192.168.1.1"` 是 **有效** IP 地址，但是 `"0.011.255.245"`、`"192.168.1.312"` 和 `"192.168@1.1"` 是 **无效** IP 地址。

给定一个只包含数字的字符串 `s` ，用以表示一个 IP 地址，返回所有可能的**有效 IP 地址**，这些地址可以通过在 `s` 中插入 `'.'` 来形成。你 **不能** 重新排序或删除 `s` 中的任何数字。你可以按 **任何** 顺序返回答案。

**示例 1：**

**输入：** s = "25525511135"

**输出：** ["255.255.11.135","255.255.111.35"]

**示例 2：**

**输入：** s = "0000"

**输出：** ["0.0.0.0"]

**示例 3：**

**输入：** s = "101023"

**输出：** ["1.0.10.23","1.0.102.3","10.1.0.23","10.10.2.3","101.0.2.3"]

**提示：**

- `1 <= s.length <= 20`
- `s` 仅由数字组成
**复杂度分析**

- 时间复杂度：O(3^4)，最多递归 4 层，每层最多 3 个分支
- 空间复杂度：O(4)，递归栈深度
