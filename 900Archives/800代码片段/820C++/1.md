## 描述

让变量 i 跟随 j 的值变化而变化
## 内容

```cpp
int main() {
    int i = 0;
    for (int j = i; i < 3;i++, j++) {
        std::cout << "i = " << i  << "  " << "j = " << j << std::endl;
    }
    return 0;
}

```

错误代码：

```cpp

int main() {
    int i = 0;
    for (int j = i; i < 3;i++) {
        std::cout << "i = " << i  << "  " << "j = " << j << std::endl;
    }
    return 0;
}


int main() {
    int i = 0;
    for (int j = i; i < 3;j = i, i++) {
        std::cout << "i = " << i  << "  " << "j = " << j << std::endl;
    }
    return 0;
}

```
