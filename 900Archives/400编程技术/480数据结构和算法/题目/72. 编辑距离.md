## 解答

```cpp
class Solution {
  public:
    int minDistance(string word1, string word2) {
        int L1 = word1.length(), L2 = word2.length();
        vector<vector<int>> dp(L1 + 1, vector<int>(L2 + 1));
        for (int i = 0; i <= L1; i++) {
            dp[i][0] = i;
        }
        for (int j = 0; j <= L2; j++) {
            dp[0][j] = j;
        }
        for (int i = 1; i <= L1; i++) {
            for (int j = 1; j <= L2; j++) {
                if (word1[i - 1] == word2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    // 分别表示：替换，删除，添加第word1[i - 1]个字符
                    dp[i][j] =
                        min(dp[i - 1][j - 1], min(dp[i - 1][j], dp[i][j - 1])) + 1;
                }
            }
        }
        return dp[L1][L2];
    }
};
```

**子问题**

长度为 L1 的 word1 和长度为 L2 的 word1 的编辑距离。
