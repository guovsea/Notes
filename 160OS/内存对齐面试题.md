一个面试题：

```cpp
#include <iostream>
struct test {
    int a;        // 0
    char b;       // 4
    long long c;  // 8
    char d[10];   // 16
    int *e;       // 32
    int f;        // 40
    int g;        // 44
};

int main(int argc, char const *argv[])
{
    std::cout << &((test*)((void*)0))->a << "\n";
    printf("%d\n", &((struct test*)((void*)0))->b);          // OK
    // std::cout << &((struct test*)(nullptr))->b << "\n";   // 段错误
    std::cout << &((test*)((void*)0))->c << "\n";
    std::cout << &((test*)((void*)0))->d << "\n";
    std::cout << &((test*)((void*)0))->e << "\n";
    std::cout << &((test*)((void*)0))->f << "\n";
    std::cout << &((test*)((void*)0))->g << "\n";
    return 0;
}
```
