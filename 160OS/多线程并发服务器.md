```c
#include <arpa/inet.h>
#include <asm-generic/errno-base.h>
#include <ctype.h>
#include <errno.h>
#include <netinet/in.h>
#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <unistd.h>

#define SERV_ADDR "192.168.10.10"
#define PORT 8090
#define BUFSIZE 1024
#define MAXTHREAD 3

pthread_t tid[MAXTHREAD];

void sys_err(const char *str) {
    perror(str);
    exit(1);
}

struct client_info {
    int thread_number;
    int fd;
    struct sockaddr_in addr;
    socklen_t addrlen;
};

void *run(void *arg) {
    struct client_info *client = (struct client_info *)arg;
    char client_str[INET_ADDRSTRLEN];
    inet_ntop(AF_INET, &client->addr.sin_addr, client_str, client->addrlen);
    printf("%s:%d connected\n", client_str, ntohs(client->addr.sin_port));
    printf("thread number = %d\n", client->thread_number);
    int readn;
    char buf[BUFSIZE];
    pthread_detach(pthread_self());
    while (1) {
        if ((readn = read(client->fd, buf, BUFSIZE)) > 0) {
            for (int i = 0; i < readn; i++) {
                buf[i] = toupper(buf[i]);
            }
            write(client->fd, buf, readn);
        } else if (readn == 0) {
            printf("%s:%d disconnected\n", client_str,
                   ntohs(client->addr.sin_port));            
                   break;
        } else {
            sys_err("read");
        }
    }
    tid[client->thread_number] = 0;
    close(client->fd);
    return 0;
}

int main(int argc, char *argv[]) {
    int listenfd;

    if ((listenfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
        sys_err("listen");
    }

    struct sockaddr_in serv_addr;
    serv_addr.sin_family = AF_INET;
    inet_pton(AF_INET, SERV_ADDR, &serv_addr.sin_addr);
    serv_addr.sin_port = htons(PORT);
    if ((bind(listenfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr))) <
        0) {
        sys_err("bind");
    }

    listen(listenfd, 20);

    struct client_info clients[MAXTHREAD];
    int n = 0;
    int connfd;
    struct sockaddr_in client_addr;
    socklen_t client_len;
    printf("Accepting connection ...\n");

    while (1) {
        client_len = sizeof(client_addr);
        if ((connfd = accept(listenfd, (struct sockaddr *)&client_addr,
                             &client_len)) < 0) {
            if (errno == EAGAIN) {
                continue;
            }
            sys_err("accept");
        }
        n = 0;
        while (tid[n] != 0) {
            n++;            
            if (n == MAXTHREAD) {
                printf("超过最大线程数量,直接退出.(实际上应该continue))\n");
                exit(1);
            }
        }
        clients[n].thread_number = n;
        clients[n].fd = connfd;
        clients[n].addr = client_addr;
        clients[n].addrlen = client_len;
        pthread_create(&tid[n], NULL, run, (void *)&clients[n]);
        n++;
    }

    return EXIT_SUCCESS;
}
```


整体思路就是用一个结构体数组保存保存的客户端的信息：
```c
struct client_info {
    int thread_number;
    int fd;
    struct sockaddr_in addr;
    socklen_t addrlen;
};
```
每调用一次 accept 将 client 的信息放到数组中某个位置上，然后将这个数组元素的指针作为参数传递给线程。

**几点注意**

1. 线程的工作函数是`void* (void*)`类型的
2. 需要有某种策略来解决：当某个连接释放后`client_info`数组和`pthread_t`数组会产生空位，如何复用这些空位。**计数器 n 从 0 递增，复用第一个空位**
```c
n = 0;
while (tid[n] != 0) {
    n++;            
    if (n == MAXTHREAD) {
        printf("超过最大线程数量,直接退出.(实际上应该continue))\n");
        exit(1);
    }
}
// n 即为第一个空位
```
5. 标志位用的 `pid[] == 0`表示空位，但实际上这种方法有些不妥。Linux 上的 pid 定义为`typedef unsigned long pthread_t`，但其他系统可能是结构体。
4. 线程处理的连接断开后记得要对标志位进行复位
