## 描述

变参模板函数的递归解析
## 内容

```cpp
template <class T>
void ToYaml(YAML::Node &node, const std::string &key, const T &val) {
    node[key] = val;
}

inline void ToYaml(YAML::Node &node) {}

template <typename T, typename... Args>
void ToYaml(YAML::Node &node, const std::string &key, const T &val,
            Args &&...args) {
    ToYaml(node, key, val);
    ToYaml(node, args...);
}
```
